  <behavior>
    - search(query, options) tries chromaSearch first when isChromaAvailable()=true
    - On chromaSearch failure or unavailable: falls back to db.fts5Search(query)
    - On fts5Search failure or empty: falls back to db.sqlSearch(query)
    - options.skipChroma=true forces skip to FTS5 (for testing tier 2 directly)
    - options.skipFts5=true forces skip to SQL (for testing tier 3 directly)
    - Returns normalized array: [{id, name, type, score}] — same shape regardless of tier used
    - Logs which tier was used to stderr for debugging
  </behavior>
  <action>
    In worker/query-engine.js, add the search() export:

    Import at top: import { chromaSearch, isChromaAvailable } from './chroma-sync.js';

    export async function search(query, options = {}) {
      const limit = options.limit || 20;

      // Tier 1: ChromaDB semantic search
      if (!options.skipChroma && isChromaAvailable()) {
        try {
          const results = await chromaSearch(query, limit);
          process.stderr.write('[search] tier=chroma results=' + results.length + '\n');
          return results.map(r => ({ id: r.id, name: r.document, type: r.metadata.type || 'unknown', score: r.score }));
        } catch (err) {
          process.stderr.write('[search] chroma failed, falling back to FTS5: ' + err.message + '\n');
        }
      }

      // Tier 2: FTS5 keyword search
      if (!options.skipFts5) {
        try {
          const results = db.fts5Search(query, limit);
          if (results.length > 0) {
            process.stderr.write('[search] tier=fts5 results=' + results.length + '\n');
            return results.map(r => ({ id: String(r.id), name: r.name, type: r.type || 'service', score: 1 }));
          }
        } catch (err) {
          process.stderr.write('[search] fts5 failed, falling back to SQL: ' + err.message + '\n');
        }
      }

      // Tier 3: Direct SQL filter (always available)
      const results = db.sqlSearch(query, limit);
      process.stderr.write('[search] tier=sql results=' + results.length + '\n');
      return results.map(r => ({ id: String(r.id), name: r.name, type: r.type || 'service', score: 0.5 }));
    }
