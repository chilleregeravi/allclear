<background>
This plan ships the read + promote side. Plan 119-01 handles writes (shadow-scan).

Per RESEARCH §3: `fs.rename` is atomic on POSIX when source and destination share a filesystem. Both shadow and live DBs sit as siblings under `projectHashDir(...)` → same filesystem by construction. **WAL sidecars** (`-wal`, `-shm`) MUST be renamed alongside the main file or SQLite sees a stale log on next open and may corrupt the DB. The shadow QE was closed by 119-01's `/scan-shadow` handler in a `finally` block, which means a clean WAL checkpoint already happened — we just need to rename the sidecar files if they exist (some SQLite versions leave empty sidecars after close).

Per RESEARCH §3: cached LIVE QueryEngine MUST be evicted from `pool` BEFORE the rename — otherwise `pool.get(projectRoot)._db` holds an fd to the renamed-out inode and subsequent calls write to the wrong file. New `evictLiveQueryEngine(projectRoot)` helper does this in one call.

Per RESEARCH §3: backups are NEVER auto-deleted. Operators clean up manually. Documented in `commands/promote-shadow.md`.

Per RESEARCH §4: `/arcanon:diff --shadow` is implemented as a courtesy extension to Phase 115's `cmdDiff`. If Phase 115's signature lands as Shape A (preferred), wiring is trivial. If a different shape, fall back to a standalone minimal-diff implementation (~50 LoC, surface in SUMMARY).
</background>

       ```javascript
       /**
        * Drop the cached LIVE QueryEngine for a project, closing its DB handle.
        * Required BEFORE /arcanon:promote-shadow renames the live DB —
        * otherwise the worker holds an fd to a renamed-out inode.
        *
        * Idempotent — returns false if no entry was cached.
        *
        * @param {string} projectRoot
        * @returns {boolean} true if an entry was evicted, false otherwise.
        */
       export function evictLiveQueryEngine(projectRoot) {
         if (!projectRoot) return false;
         const qe = pool.get(projectRoot);
         if (!qe) return false;
         try { qe._db.close(); } catch { /* already closed */ }
         pool.delete(projectRoot);
         return true;
       }
       ```

       ```javascript
       async function cmdPromoteShadow(flags) {
         const projectRoot = process.cwd();
         const dir = projectHashDir(projectRoot);
         const livePath = path.join(dir, 'impact-map.db');
         const shadowPath = path.join(dir, 'impact-map-shadow.db');

         // Silent in non-Arcanon directory (mirrors NAV-01)
         if (!fs.existsSync(livePath) && !fs.existsSync(shadowPath)) {
           process.exit(0);
         }

         // Shadow MUST exist
         if (!fs.existsSync(shadowPath)) {
           process.stderr.write('error: no shadow DB to promote (looked for ' + shadowPath + ')\n');
           process.exit(2);
         }

         // Evict cached live QE BEFORE any rename — see threat T-119-02-01
         const evicted = evictLiveQueryEngine(projectRoot);

         const ts = new Date().toISOString().replace(/[:.]/g, '-');
         let backupPath = null;

         try {
           if (fs.existsSync(livePath)) {
             // Step 1: backup live (main + sidecars)
             backupPath = livePath + '.pre-promote-' + ts;
             fs.renameSync(livePath, backupPath);
             for (const sfx of ['-wal', '-shm']) {
               if (fs.existsSync(livePath + sfx)) {
                 fs.renameSync(livePath + sfx, backupPath + sfx);
               }
             }
           }

           // Step 2: promote shadow → live (main + sidecars)
           fs.renameSync(shadowPath, livePath);
           for (const sfx of ['-wal', '-shm']) {
             if (fs.existsSync(shadowPath + sfx)) {
               fs.renameSync(shadowPath + sfx, livePath + sfx);
             }
           }
         } catch (err) {
           // Rollback attempt: if shadow→live failed but backup was made, restore live from backup.
           if (backupPath && fs.existsSync(backupPath) && !fs.existsSync(livePath)) {
             try {
               fs.renameSync(backupPath, livePath);
               for (const sfx of ['-wal', '-shm']) {
                 if (fs.existsSync(backupPath + sfx)) fs.renameSync(backupPath + sfx, livePath + sfx);
               }
             } catch { /* ignore — surfaced via thrown error below */ }
           }
           process.stderr.write('error: promote failed: ' + err.message + '\n');
           process.exit(1);
         }

         // Step 3: report
         const human = backupPath
           ? 'Promoted shadow → live.\nBackup: ' + backupPath + '\nDelete backup manually when no longer needed.'
           : 'No live DB to back up; shadow promoted to live.\nLive: ' + livePath;
         emit(
           { ok: true, backup_path: backupPath, live_path: livePath, evicted_cached_qe: evicted },
           flags,
           human,
         );
       }
       ```

       ```javascript
         const liveSvcs = qeA._db.prepare("SELECT name, type, language FROM services").all();
         const shadSvcs = qeB._db.prepare("SELECT name, type, language FROM services").all();
         const liveByName = new Map(liveSvcs.map(s => [s.name, s]));
         const shadByName = new Map(shadSvcs.map(s => [s.name, s]));

         const services = { added: [], removed: [], modified: [] };
         for (const s of shadSvcs) {
           if (!liveByName.has(s.name)) services.added.push(s);
           else {
             const live = liveByName.get(s.name);
             if (live.type !== s.type || live.language !== s.language) services.modified.push({ name: s.name, before: live, after: s });
           }
         }
         for (const s of liveSvcs) {
           if (!shadByName.has(s.name)) services.removed.push(s);
         }

         // Same shape for connections — compare on (source_service_id, target_service_id, protocol, path)
         const connsA = qeA._db.prepare("SELECT source_service_id, target_service_id, protocol, path FROM connections").all();
         const connsB = qeB._db.prepare("SELECT source_service_id, target_service_id, protocol, path FROM connections").all();
         const keyOf = c => `${c.source_service_id}→${c.target_service_id}|${c.protocol}|${c.path}`;
         const liveByKey = new Map(connsA.map(c => [keyOf(c), c]));
         const shadByKey = new Map(connsB.map(c => [keyOf(c), c]));

         const connections = { added: [], removed: [], modified: [] };
         for (const c of connsB) if (!liveByKey.has(keyOf(c))) connections.added.push(c);
         for (const c of connsA) if (!shadByKey.has(keyOf(c))) connections.removed.push(c);

         return {
           services,
           connections,
           summary: {
             services: { added: services.added.length, removed: services.removed.length, modified: services.modified.length },
             connections: { added: connections.added.length, removed: connections.removed.length, modified: 0 },
           },
         };
       }

       function formatDiffHuman(r) {
         return [
           `Services: +${r.summary.services.added} added, -${r.summary.services.removed} removed, ~${r.summary.services.modified} modified`,
           `Connections: +${r.summary.connections.added} added, -${r.summary.connections.removed} removed`,
         ].join('\n');
       }
       ```
