<background>
Phase 121 is the consumption-side counterpart to Phase 120's data-layer ship. Plan 121-01 is the foundation: the schema column, the loader, the match logic, and the wire-in point in `manager.js`. Plan 121-02 layers user extension + UI surfacing on top.

**Why per-repo, not per-service**: actors live in a global table joined to services via `actor_connections`. Labeling is a property of the actor, not a property of the service. Running once per repo (after the per-service loop) means: (a) one DB write per actor regardless of how many services connect to it, (b) clear log line per repo for `slog('INFO', 'actor-labeling done', ...)`, (c) no coupling to the per-service enricher contract (which returns `Record<key, value>` for `node_metadata` writes — actors are not in `node_metadata`).

**Migration 018 sequencing**: Migration 016 (Phase 113) is current head. Phase 117 (this milestone, Wave 3) introduces 017 (`scan_overrides`). Phase 121 (Wave 5) is downstream of Phase 117 in time but logically independent. Migration 018 is the next free integer. If Phase 117 slips, this plan must move to 017 — see §11 fallback in RESEARCH.

**Catalog loaded once per scan, not per actor**: `manager.js` calls `loadShippedCatalog()` once at scan-start (or once per scanRepos invocation), passes the normalized catalog into `runActorLabeling` for each repo. Module-level cache inside `externals-catalog.js` makes repeat calls cheap.

**Load-once semantics (cache invalidation)**: The catalog is cached at module level for the process lifetime. Restarting the worker re-reads the file. There's no `--reload-catalog` flag in this plan; that's a future enhancement if user demand surfaces.
</background>

/**
 * worker/scan/enrichment/externals-catalog.js — Phase 121 / INT-06.
 *
 * Loads the shipped externals catalog (data/known-externals.yaml) and
 * exposes a pure matchActor(actorName, catalog) function used by the
 * actor labeling pass.
 *
 * The loader is module-cached for the process lifetime — restart worker
 * to pick up catalog file changes.
 */

import fs from "node:fs";
import path from "node:path";
import { fileURLToPath } from "node:url";
import yaml from "js-yaml";

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

// Module-level cache keyed by absolute path
const _cache = new Map();

function defaultShippedPath() {
  // worker/scan/enrichment/externals-catalog.js -> ../../../data/known-externals.yaml
  return path.resolve(__dirname, "..", "..", "..", "data", "known-externals.yaml");
}

/**
 * @typedef {{ label: string, hosts: string[], ports: number[] }} CatalogEntry
 * @typedef {{ entries: Map<string, CatalogEntry> }} NormalizedCatalog
 */

/**
 * @param {string} [explicitPath]
 * @param {{ warn?: Function } | null} [logger]
 * @returns {NormalizedCatalog}
 */
export function loadShippedCatalog(explicitPath, logger = null) {
  const absPath = path.resolve(explicitPath || defaultShippedPath());
  if (_cache.has(absPath)) return _cache.get(absPath);

  if (!fs.existsSync(absPath)) {
    logger?.warn?.(`externals-catalog: file not found at ${absPath} — no labels will be assigned`);
    const empty = { entries: new Map() };
    _cache.set(absPath, empty);
    return empty;
  }

  let raw;
  try {
    const text = fs.readFileSync(absPath, "utf8");
    raw = yaml.load(text);
  } catch (err) {
    logger?.warn?.(`externals-catalog: parse error at ${absPath}: ${err.message}`);
    const empty = { entries: new Map() };
    _cache.set(absPath, empty);
    return empty;
  }

  const normalized = normalizeCatalog(raw, logger);
  _cache.set(absPath, normalized);
  return normalized;
}

/** Test-only — clear the module cache. */
export function _clearCatalogCache() {
  _cache.clear();
}

function normalizeCatalog(raw, logger) {
  const entries = new Map();
  if (!raw || typeof raw !== "object") return { entries };
  const rawEntries = raw.entries;
  if (!rawEntries) return { entries };

  let pairs;
  if (Array.isArray(rawEntries)) {
    pairs = rawEntries
      .filter((e) => e && typeof e === "object" && typeof e.id === "string")
      .map((e) => [e.id, e]);
  } else if (typeof rawEntries === "object") {
    pairs = Object.entries(rawEntries);
  } else {
    return { entries };
  }

  for (const [slug, entry] of pairs) {
    if (!entry || typeof entry !== "object") {
      logger?.warn?.(`externals-catalog: skipping non-object entry "${slug}"`);
      continue;
    }
    const label = entry.label;
    if (typeof label !== "string" || label.trim() === "") {
      logger?.warn?.(`externals-catalog: skipping entry "${slug}" — missing or empty label`);
      continue;
    }
    const hosts = Array.isArray(entry.hosts) ? entry.hosts.filter((h) => typeof h === "string") : [];
    const ports = Array.isArray(entry.ports) ? entry.ports.filter((p) => Number.isInteger(p)) : [];
    if (hosts.length === 0 && ports.length === 0) {
      logger?.warn?.(`externals-catalog: skipping entry "${slug}" — no hosts or ports`);
      continue;
    }
    entries.set(slug, { label: label.trim(), hosts, ports });
  }

  return { entries };
}

/**
 * @param {string} actorName
 * @param {NormalizedCatalog} catalog
 * @returns {string | null}
 */
export function matchActor(actorName, catalog) {
  if (typeof actorName !== "string" || actorName.length === 0) return null;
  if (!catalog || !(catalog.entries instanceof Map)) return null;

  let hostname = null;
  if (/^[a-z][a-z0-9+.-]*:\/\//i.test(actorName)) {
    try {
      hostname = new URL(actorName).hostname.toLowerCase();
    } catch {}
  } else {
    hostname = actorName.split("/")[0].split(":")[0].toLowerCase();
  }

  if (hostname) {
    for (const entry of catalog.entries.values()) {
      for (const pattern of entry.hosts) {
        if (matchHost(hostname, pattern)) return entry.label;
      }
    }
  }

  const portMatch = actorName.match(/:(\d+)(?:\/|$)/);
  if (portMatch) {
    const port = parseInt(portMatch[1], 10);
    for (const entry of catalog.entries.values()) {
      if (entry.ports.includes(port)) return entry.label;
    }
  }

  return null;
}

function matchHost(hostname, pattern) {
  const p = pattern.toLowerCase();
  if (!p.includes("*")) return hostname === p;
  if (!p.startsWith("*.")) return false;
  const suffix = p.slice(2);
  if (hostname === suffix) return false; // bare host does NOT match *.suffix
  return hostname.endsWith("." + suffix);
}

    1. Create `plugins/arcanon/worker/scan/enrichment/actor-labeler.js`:

```javascript
/**
 * worker/scan/enrichment/actor-labeler.js — Phase 121 / INT-06.
 *
 * Per-repo actor labeling pass. Invoked from manager.js once per repo,
 * after the per-service enrichment loop and BEFORE the
 * slog('INFO', 'enrichment done', ...) line.
 *
 * Self-healing: writes label = ? for EVERY actor connected to the repo,
 * passing NULL when no catalog entry matches. This way, removing an
 * entry from the catalog clears the corresponding actors.label on the
 * next scan rather than leaving a stale label.
 *
 * Failure-isolated: any error inside this pass is caught, logged at WARN,
 * and the scan continues. Mirrors the runEnrichmentPass posture.
 */

import { matchActor } from "./externals-catalog.js";

/**
 * @param {number} repoId
 * @param {import('better-sqlite3').Database} db
 * @param {{ warn?: Function, info?: Function } | null} logger
 * @param {import('./externals-catalog.js').NormalizedCatalog} catalog
 * @returns {Promise<{ matched: number, considered: number }>}
 */
export async function runActorLabeling(repoId, db, logger, catalog) {
  try {
    const rows = db
      .prepare(
        `SELECT DISTINCT a.id, a.name
         FROM actors a
         JOIN actor_connections ac ON ac.actor_id = a.id
         JOIN services s ON s.id = ac.service_id
         WHERE s.repo_id = ?`,
      )
      .all(repoId);

    if (rows.length === 0) return { matched: 0, considered: 0 };

    const stmt = db.prepare("UPDATE actors SET label = ? WHERE id = ?");
    let matched = 0;
    for (const row of rows) {
      const label = matchActor(row.name, catalog);
      stmt.run(label, row.id);
      if (label !== null) matched++;
    }
    return { matched, considered: rows.length };
  } catch (err) {
    logger?.warn?.(`actor-labeling failed for repo ${repoId}: ${err.message}`);
    return { matched: 0, considered: 0 };
  }
}
```
