<!-- logger.log signature (from worker/lib/logger.js) -->
```javascript
// createLogger returns an object with:
log(level, msg, extra = {})
// level: 'DEBUG' | 'INFO' | 'WARN' | 'ERROR'
// Writes JSON: { ts, level, msg, ...extra }

// httpLog helper already in http.js wraps log():
function httpLog(level, msg, extra = {}) {
  if (log) log.log(level, msg, { component: 'http', ...extra });
}
```
