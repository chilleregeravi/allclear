    # tests/fixtures/impact-hook/setup-fake-db.sh
    # Creates a deterministic fake impact-map.db with:
    #   - repo at $PROJECT_ROOT

    #   setup_fake_db

    setup_fake_db() {
      [[ -z "${ARCANON_DATA_DIR:-}" ]] && { echo "ARCANON_DATA_DIR not set" >&2; return 1; }
      [[ -z "${PROJECT_ROOT:-}" ]] && { echo "PROJECT_ROOT not set" >&2; return 1; }

      mkdir -p "$PROJECT_ROOT/services/auth"
      mkdir -p "$PROJECT_ROOT/services/auth-legacy"
      mkdir -p "$PROJECT_ROOT/services/web"

      local db_dir="${ARCANON_DATA_DIR}/projects/${db_hash}"

    CREATE TABLE services(id INTEGER PRIMARY KEY, repo_id INTEGER, name TEXT, root_path TEXT, language TEXT);

    INSERT INTO services VALUES(2,1,'web','$PROJECT_ROOT/services/web','node');

    SQL

      echo "$db_path"

    ```bash
    #!/usr/bin/env bash
    # tests/impact-hook-latency.sh — 100-iteration p99 benchmark for impact-hook.sh
    # Exits 0 if p99 < THRESHOLD_MS, 1 otherwise.
    #
    # Usage:
    #   ARCANON_DATA_DIR=/tmp/x PROJECT_ROOT=/tmp/x/proj THRESHOLD_MS=50 bash impact-hook-latency.sh

    set -u
    THRESHOLD_MS="${THRESHOLD_MS:-50}"
    ITERATIONS="${ITERATIONS:-100}"
    REPO_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
    SCRIPT="${REPO_ROOT}/plugins/arcanon/scripts/impact-hook.sh"
    FILE="${PROJECT_ROOT}/services/auth/index.js"
    INPUT=$(printf '{"tool_name":"Write","tool_input":{"file_path":"%s"}}' "$FILE")

    declare -a TIMES
    for i in $(seq 1 "$ITERATIONS"); do
      t0=$(date +%s%3N 2>/dev/null || python3 -c 'import time;print(int(time.time()*1000))')
      printf '%s' "$INPUT" | bash "$SCRIPT" >/dev/null 2>&1
      t1=$(date +%s%3N 2>/dev/null || python3 -c 'import time;print(int(time.time()*1000))')
      TIMES+=($(( t1 - t0 )))
    done

    # Sort ascending; p99 = element at index floor(0.99 * N) (99th of 100 = index 98)
    SORTED=($(printf '%s\n' "${TIMES[@]}" | sort -n))
    P99_INDEX=$(( (ITERATIONS * 99 / 100) - 1 ))
    [[ "$P99_INDEX" -lt 0 ]] && P99_INDEX=0
    P99="${SORTED[$P99_INDEX]}"

    echo "impact-hook latency: iterations=${ITERATIONS} p99=${P99}ms threshold=${THRESHOLD_MS}ms"
    if (( P99 > THRESHOLD_MS )); then
      echo "FAIL: p99 ${P99}ms exceeds threshold ${THRESHOLD_MS}ms" >&2
      exit 1
    fi
    exit 0
    ```
